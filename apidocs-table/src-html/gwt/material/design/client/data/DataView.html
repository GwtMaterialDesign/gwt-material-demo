020package gwt.material.design.client.data;
021
022import com.google.gwt.dom.client.Element;
023import com.google.gwt.user.client.ui.Panel;
024import com.google.gwt.view.client.HasKeyProvider;
025import com.google.gwt.view.client.HasRows;
026import gwt.material.design.client.data.component.CategoryComponent;
027import gwt.material.design.client.data.component.Component;
028import gwt.material.design.client.data.component.ComponentFactory;
029import gwt.material.design.client.data.component.Components;
030import gwt.material.design.client.data.component.RowComponent;
031import gwt.material.design.client.data.factory.CategoryComponentFactory;
032import gwt.material.design.client.data.factory.RowComponentFactory;
033import gwt.material.design.client.js.JsTableSubHeaders;
034import gwt.material.design.client.ui.MaterialProgress;
035import gwt.material.design.client.ui.table.TableHeader;
036import gwt.material.design.client.ui.table.TableScaffolding;
037import gwt.material.design.client.ui.table.cell.Column;
038
039import java.util.List;
040
041/**
042 * A data view is in charge of rendering the data within the data table.
043 *
044 * @author Ben Dol
045 */
046public interface DataView<T> extends HasRows, HasKeyProvider<T> {
047
048    /**
049     * Render the data view components to DOM.
050     */
051    void render(Components<Component<?>> components);
052
053    /**
054     * Setup the data view.
055     */
056    void setup(TableScaffolding scaffolding) throws Exception;
057
058    /**
059     * Destroy the data view.
060     */
061    void destroy();
062
063    /**
064     * Data views loader has loaded data for use.
065     */
066    void loaded(int startIndex, List<T> data);
067
068    /**
069     * Invoke a refresh on the dataview.
070     */
071    void refreshView();
072
073    /**
074     * Set a {@link DataSource} for data loading.
075     */
076    void setDataSource(DataSource<T> dataSource);
077
078    /**
079     * Get the configured data source.
080     */
081    DataSource<T> getDataSource();
082
083    /**
084     * Set your own custom data view renderer.
085     * See {@link BaseRenderer} for the standard rendering.
086     */
087    void setRenderer(Renderer<T> renderer);
088
089    /**
090     * Get the maximum row height.
091     */
092    int getRowHeight();
093
094    /**
095     * Set maximum row height. The minimum row height is
096     * 33 pixels as a material design specification.
097     */
098    void setRowHeight(int rowHeight);
099
100    /**
101     * Get the displays main container.
102     */
103    Panel getContainer();
104
105    /**
106     * Get the data views id.
107     */
108    String getViewId();
109
110    /**
111     * Update a models row within the table.
112     * @param model a model with a valid <code>equals</code> method.
113     */
114    void updateRow(T model);
115
116    /**
117     * Get a row by its representing model.
118     * @param model the model assigned to a row.
119     */
120    RowComponent<T> getRow(T model);
121
122    /**
123     * Get a row by its rendered index.
124     * @param index the value of the render index.
125     */
126    RowComponent<T> getRow(int index);
127
128    /**
129     * Get a models {@link RowComponent} or null if not found.
130     * @param model a model with a valid <code>equals</code> method.
131     * @return the models representing {@link RowComponent}.
132     */
133    RowComponent<T> getRowByModel(T model);
134
135    /**
136     * Clear data rows.
137     *
138     * @param clearData should we also clear the stored data.
139     */
140    void clearRows(boolean clearData);
141
142    /**
143     * Clear all rows and categories.
144     *
145     * @param clearData should we also clear the stored data.
146     */
147    void clearRowsAndCategories(boolean clearData);
148
149    /**
150     * Clear all categories.
151     */
152    void clearCategories();
153
154    /**
155     * Set the data views display view.
156     */
157    void setDisplay(DataView<T> display);
158
159    /**
160     * Check if a header with the given index is visible.
161     */
162    boolean isHeaderVisible(int colIndex);
163
164    /**
165     * Get the list of rendered header widgets.
166     */
167    List<TableHeader> getHeaders();
168
169    /**
170     * Add a new column to the data view.
171     *
172     * @param column the column object
173     */
174    void addColumn(Column<T, ?> column);
175
176    /**
177     * Add a new column to the data view.
178     *
179     * @param column the column object
180     */
181    void addColumn(Column<T, ?> column, String header);
182
183    /**
184     * Inserts a column into the table at the specified index.
185     *
186     * @param beforeIndex the index to insert the column
187     * @param col the column to be added
188     * @throws IndexOutOfBoundsException if the index is out of range
189     */
190    void insertColumn(int beforeIndex, Column<T, ?> col, String header);
191
192    /**
193     * Remove an existing column by index.
194     * @param colIndex the columns placement index
195     */
196    void removeColumn(int colIndex);
197
198    /**
199     * Get the current defined columns.
200     */
201    List<Column<T, ?>> getColumns();
202
203    /**
204     * Get the offset column index (used when selection mode is on).
205     */
206    int getColumnOffset();
207
208    /**
209     * Set the data views row selection type.
210     */
211    void setSelectionType(SelectionType selectionType);
212
213    /**
214     * Get the data views row selection type.
215     */
216    SelectionType getSelectionType();
217
218    /**
219     * Enable the use of sticky table header bar.
220     */
221    void setUseStickyHeader(boolean stickyHeader);
222
223    /**
224     * True if we are using sticky table header.
225     */
226    boolean isUseStickyHeader();
227
228    /**
229     * Get the subheader library API.
230     */
231    JsTableSubHeaders getSubheaderLib();
232
233    /**
234     * Select or deselect all rows.
235     * @param select true will select, false with deselect
236     */
237    void selectAllRows(boolean select);
238
239    /**
240     * Select or deselect all rows.
241     *
242     * @param select true will select, false with deselect
243     * @param fireEvent fire the '{@value gwt.material.design.client.ui.table.TableEvents#SELECT_ALL}' event.
244     */
245    void selectAllRows(boolean select, boolean fireEvent);
246
247    /**
248     * Select a row by given element.
249     *
250     * @param row element of the row selection
251     */
252    void toggleRowSelect(Element row);
253
254    /**
255     * Select a row by given element.
256     *
257     * @param row element of the row selection
258     * @param fireEvent fire the '{@value gwt.material.design.client.ui.table.TableEvents#ROW_SELECT}' event.
259     */
260    void toggleRowSelect(Element row, boolean fireEvent);
261
262    /**
263     * Select a given row.
264     */
265    void selectRow(Element row, boolean fireEvent);
266
267    /**
268     * Unselect a selected row.
269     */
270    void deselectRow(Element row, boolean fireEvent);
271
272    /**
273     * Does this view have unselected rows.
274     *
275     * @param visibleOnly should we restrict this check to visible rows only.
276     * @return true if there are unselected rows in this view.
277     */
278    boolean hasDeselectedRows(boolean visibleOnly);
279
280    /**
281     * Does this view have selected rows.
282     *
283     * @param visibleOnly should we restrict this check to visible rows only.
284     * @return true if there are selected rows in this view.
285     */
286    boolean hasSelectedRows(boolean visibleOnly);
287
288    /**
289     * Get all the selected row models.
290     * @param visibleOnly should we restrict this check to visible rows only.
291     * @return list of the currently selected row models.
292     */
293    List<T> getSelectedRowModels(boolean visibleOnly);
294
295    /**
296     * Is the data view setup.
297     */
298    boolean isSetup();
299
300    /**
301     * Set values associated with the rows in the visible range.
302     *
303     * @param start the start index of the data
304     * @param values the values within the range
305     */
306    void setRowData(int start, List<? extends T> values);
307
308    /**
309     * Get the visible item count of the data view.
310     */
311    int getVisibleItemCount();
312
313    /**
314     * Get a stored category component.
315     * @param categoryName name of the category component.
316     */
317    CategoryComponent getCategory(String categoryName);
318
319    /**
320     * Get all registered category components.
321     */
322    List<CategoryComponent> getCategories();
323
324    /**
325     * Get all the open {@link CategoryComponent}'s.
326     */
327    List<CategoryComponent> getOpenCategories();
328
329    /**
330     * Check if a category has data to provide.
331     */
332    boolean isCategoryEmpty(CategoryComponent category);
333
334    /**
335     * Explicitly add a category, which will be drawn to the table.
336     * If the category already exists then it will be ignored.
337     * @param category The category name.
338     */
339    void addCategory(String category);
340
341    /**
342     * Explicitly add a {@link CategoryComponent}, which will be drawn to the table.
343     * If the category already exists then it will be ignored.
344     * @param category The category data defined.
345     */
346    void addCategory(CategoryComponent category);
347
348    /**
349     * Has this data view got an existing {@link CategoryComponent} with given name.
350     */
351    boolean hasCategory(String categoryName);
352
353    /**
354     * Disable a data view category.
355     */
356    void disableCategory(String categoryName);
357
358    /**
359     * Open an existing Category.
360     */
361    void openCategory(String categoryName);
362
363    /**
364     * Open an existing Category.
365     */
366    void openCategory(CategoryComponent category);
367
368    /**
369     * Close an existing Category.
370     */
371    void closeCategory(String categoryName);
372
373    /**
374     * Close an existing Category.
375     */
376    void closeCategory(CategoryComponent category);
377
378    /**
379     * Get the data views current sort context, or null if no sort context is applied.
380     */
381    SortContext<T> getSortContext();
382
383    /**
384     * Is redrawing the data view.
385     */
386    boolean isRedraw();
387
388    /**
389     * Set redraw data view elements.
390     */
391    void setRedraw(boolean redraw);
392
393    /**
394     * Set your own custom {@link RowComponentFactory} to generate your row components.
395     */
396    void setRowFactory(RowComponentFactory<T> rowFactory);
397
398    /**
399     * Set your own custom {@link CategoryComponentFactory} to generate your categories.
400     */
401    void setCategoryFactory(ComponentFactory<? extends CategoryComponent, String> categoryFactory);
402
403    /**
404     * Set the data views loading mask.
405     */
406    void setLoadMask(boolean loadMask);
407
408    /**
409     * Is the data views loading mask applied.
410     */
411    boolean isLoadMask();
412
413    /**
414     * Get the data views progress widget.
415     */
416    MaterialProgress getProgressWidget();
417
418    /**
419     * Get the configured total rows count.
420     */
421    int getTotalRows();
422
423    /**
424     * Set the total row count.
425     */
426    void setTotalRows(int totalRows);
427
428    /**
429     * Is the data view using categories where possible.
430     */
431    boolean isUseCategories();
432
433    /**
434     * Allow the data view to generate categories,
435     * or use added categories for row data.
436     */
437    void setUseCategories(boolean useCategories);
438
439    /**
440     * Is the data view using loading overlay mask.
441     */
442    boolean isUseLoadOverlay();
443
444    /**
445     * Use the loading overlay mask.
446     */
447    void setUseLoadOverlay(boolean useLoadOverlay);
448
449    /**
450     * Is the data view using row expansion.
451     */
452    boolean isUseRowExpansion();
453
454    /**
455     * Use row expansion functionality, giving the user
456     * a way to expand rows for extra information.
457     */
458    void setUseRowExpansion(boolean useRowExpansion);
459
460    /**
461     * Get the long press duration requirement.
462     */
463    int getLongPressDuration();
464
465    /**
466     * Set the long press duration requirement.
467     */
468    void setLongPressDuration(int longPressDuration);
469
470    /**
471     * Set the table body height.
472     */
473    void setHeight(String height);
474
475    /**
476     * Get the table body height.
477     */
478    String getHeight();
479}
