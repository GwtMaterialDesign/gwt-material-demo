020package gwt.material.design.client.data.component;
021
022import java.util.ArrayList;
023import java.util.Collection;
024
025/**
026 * @author Ben Dol
027 */
028public class Components<T extends Component<?>> extends ArrayList<T> {
029
030    public interface Clone<T> {
031        T clone(T clone);
032    }
033
034    private int maxSize = -1;
035
036    public Components() {
037    }
038
039    public Components(int maxSize) {
040        this.maxSize = maxSize;
041    }
042
043    public Components(Collection<? extends T> components) {
044        this(components, null);
045    }
046
047    public Components(Collection<? extends T> components, Clone<T> clone) {
048        super(components);
049
050        if(clone != null) {
051            Components<T> cloned = new Components<>();
052            for(T component : this) {
053                cloned.add(clone.clone(component));
054            }
055            super.clear();
056            addAll(cloned);
057        }
058    }
059
060    /**
061     * Clear both the Widget and Component.
062     */
063    @Override
064    public void clear() {
065        clearWidgets();
066        super.clear();
067    }
068
069    /**
070     * Clear the components Widgets, without dumping the component.
071     */
072    public void clearWidgets() {
073        // clear dom rows
074        for(T component : this) {
075            component.clearWidget();
076        }
077    }
078
079    /**
080     * Clear the component data without touching the DOM.
081     */
082    public void clearComponents() {
083        super.clear();
084    }
085
086    public boolean isFull() {
087        return maxSize != -1 && size() == maxSize;
088    }
089
090    @Override
091    public boolean add(T t) {
092        return !isFull() && super.add(t);
093    }
094
095    @Override
096    public boolean addAll(Collection<? extends T> c) {
097        return canAdd(c) && super.addAll(c);
098    }
099
100    @Override
101    public boolean addAll(int index, Collection<? extends T> c) {
102        return canAdd(c) && super.addAll(index, c);
103    }
104
105    @Override
106    public T remove(int index) {
107        T component = super.get(index);
108        if(component != null) {
109            component.clearWidget();
110        }
111        return softRemove(index);
112    }
113
114    public T softRemove(int index) {
115        return super.remove(index);
116    }
117
118    public boolean softRemove(T item) {
119        int i = indexOf(item);
120        if (i == -1) {
121            return false;
122        }
123        super.remove(i);
124        return true;
125    }
126
127    public boolean canAdd(Collection<? extends T> c) {
128        return !isFull() && (maxSize == -1 || (c.size() + size()) <= maxSize);
129    }
130}
