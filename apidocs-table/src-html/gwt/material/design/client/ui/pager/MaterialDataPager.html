020package gwt.material.design.client.ui.pager;
021
022import com.google.gwt.core.client.GWT;
023import gwt.material.design.client.data.DataSource;
024import gwt.material.design.client.data.SortContext;
025import gwt.material.design.client.data.component.CategoryComponent;
026import gwt.material.design.client.data.loader.LoadCallback;
027import gwt.material.design.client.data.loader.LoadConfig;
028import gwt.material.design.client.data.loader.LoadResult;
029import gwt.material.design.client.ui.table.MaterialDataTable;
030
031import java.util.List;
032
033/**
034 * Material Data Pager - a simple pager for Material Data Table component
035 *
036 * @author kevzlou7979
037 */
038public class MaterialDataPager<T> extends MaterialDataPagerBase<T> implements HasPager {
039
040    private MaterialDataTable<T> table;
041    private DataSource<T> dataSource;
042    private int offset = 0;
043    private int limit = 0;
044    private int currentPage = 1;
045    private int totalRows = 0;
046    private int[] limitOptions = new int[]{5, 10, 20};
047
048    public MaterialDataPager(MaterialDataTable<T> table, DataSource<T> dataSource) {
049        super();
050        this.table = table;
051        this.dataSource = dataSource;
052    }
053
054    public MaterialDataPager() {
055        super();
056    }
057
058    @Override
059    protected void onLoad() {
060        super.onLoad();
061
062        initialize();
063    }
064
065    /**
066     * Initialize the data pager for navigation
067     */
068    protected void initialize() {
069        buildNumPagePanel();
070        buildLimitOptionsPanel();
071        buildActionPanel();
072
073        limit = limitOptions[0];
074        firstPage();
075        iconNext.addClickHandler(event -> next());
076        iconPrev.addClickHandler(event -> previous());
077
078        listPages.addValueChangeHandler(event -> gotoPage(event.getValue()));
079
080        // Build the limit options listbox
081        listLimitOptions.clear();
082        for (int limitOption : limitOptions) {
083            listLimitOptions.addItem(limitOption);
084        }
085        listLimitOptions.addValueChangeHandler(valueChangeEvent -> {
086            limit = valueChangeEvent.getValue();
087            if ((totalRows / currentPage) < limit) {
088                lastPage();
089                return;
090            }
091            gotoPage(listPages.getValue());
092        });
093    }
094
095    @Override
096    public void next() {
097        currentPage++;
098        gotoPage(currentPage);
099    }
100
101    @Override
102    public void previous() {
103        currentPage--;
104        gotoPage(currentPage);
105    }
106
107    @Override
108    public void lastPage() {
109        if (isExcess()) {
110            gotoPage((totalRows / limit) + 1);
111        }else {
112            gotoPage(totalRows / limit);
113        }
114
115        listPages.setSelectedIndex(currentPage - 1);
116    }
117
118    @Override
119    public void firstPage() {
120        gotoPage(1);
121    }
122
123    @Override
124    public void gotoPage(int page) {
125        this.currentPage = page;
126        doLoad((page * limit) - limit, limit);
127    }
128
129    @Override
130    public int getCurrentPage() {
131        return currentPage;
132    }
133
134    @Override
135    public void setLimit(int limit) {
136        this.limit = limit;
137    }
138
139    @Override
140    public boolean isNext() {
141        return offset + limit < totalRows;
142    }
143
144    @Override
145    public boolean isPrevious() {
146        return offset - limit >= 0;
147    }
148
149    /**
150     * Set the limit as an array of options to be populated inside the rows per page listbox
151     */
152    public void setLimitOptions(int... limitOptions) {
153        this.limitOptions = limitOptions;
154    }
155
156    /**
157     * Check whether there are excess rows to be rendered with given limit
158     */
159    protected boolean isExcess() {
160        return totalRows % limit > 0;
161    }
162
163    /**
164     * Check whether the pager is on the last currentPage.
165     */
166    protected boolean isLastPage() {
167        return currentPage == (totalRows / limit) + 1;
168    }
169
170    /**
171     * Load the datasource within a given offset and limit
172     */
173    protected void doLoad(int offset, int limit) {
174        this.offset = offset;
175        dataSource.load(new LoadConfig<T>() {
176            @Override
177            public int getOffset() {
178                return offset;
179            }
180
181            @Override
182            public int getLimit() {
183                // Check whether the pager has excess rows with given limit
184                if (isLastPage() & isExcess()) {
185                    // Get the difference between total rows and excess rows
186                    return totalRows - offset;
187                }
188                return limit;
189            }
190
191            @Override
192            public SortContext<T> getSortContext() {
193                return table.getSortContext();
194            }
195
196            @Override
197            public List<CategoryComponent> getOpenCategories() {
198                return table.getOpenCategories();
199            }
200        }, new LoadCallback<T>() {
201            @Override
202            public void onSuccess(LoadResult<T> loadResult) {
203                totalRows = loadResult.getTotalLength();
204                table.setVisibleRange(offset, limit);
205                table.loaded(loadResult.getOffset(), loadResult.getData());
206                updateUi();
207            }
208
209            @Override
210            public void onFailure(Throwable caught) {
211                GWT.log("Load failure", caught);
212                //TODO: What we need to do on failure? May be clear table?
213            }
214        });
215    }
216
217    /**
218     * Set and update the ui fields of the pager after the datasource load callback
219     */
220    protected void updateUi() {
221
222        // Action label (current selection)
223        int lastRow = (isExcess() & isLastPage()) ? totalRows : (offset + limit);
224        actionLabel.setText((offset + 1) + "-" + lastRow + " of " + totalRows);
225
226        // Build the currentPage number listbox
227        listPages.clear();
228        int pages = (isExcess()) ? (totalRows / limit) + 1 : totalRows / limit;
229        for (int i = 1; i <= pages; i++) {
230            listPages.addItem(i);
231        }
232        listPages.setSelectedIndex(currentPage - 1);
233
234        iconNext.setEnabled(true);
235        iconPrev.setEnabled(true);
236
237        if (isLastPage() || currentPage == (totalRows / limit)) {
238            iconNext.setEnabled(false);
239        }
240
241        if (!isPrevious()) {
242            iconPrev.setEnabled(false);
243        }
244    }
245
246    public MaterialDataTable<T> getTable() {
247        return table;
248    }
249
250    public void setTable(MaterialDataTable<T> table) {
251        this.table = table;
252    }
253
254    public DataSource<T> getDataSource() {
255        return dataSource;
256    }
257
258    public void setDataSource(DataSource<T> dataSource) {
259        this.dataSource = dataSource;
260    }
261}
