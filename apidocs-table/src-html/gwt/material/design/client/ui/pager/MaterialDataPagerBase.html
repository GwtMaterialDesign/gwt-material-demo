020package gwt.material.design.client.ui.pager;
021
022import com.google.gwt.dom.client.Document;
023import gwt.material.design.client.base.constants.TableCssName;
024import gwt.material.design.client.constants.IconType;
025import gwt.material.design.client.constants.WavesType;
026import gwt.material.design.client.ui.MaterialColumn;
027import gwt.material.design.client.ui.MaterialIcon;
028import gwt.material.design.client.ui.MaterialListValueBox;
029import gwt.material.design.client.ui.MaterialRow;
030import gwt.material.design.client.ui.html.Span;
031
032/**
033 * Base class implementation of MaterialDataPager which builds the User Interface of the overall pager
034 * @author kevzlou7979
035 */
036public class MaterialDataPagerBase<T> extends MaterialRow {
037
038    // Number Page Panel
039    protected MaterialColumn numPagePanel = new MaterialColumn();
040    protected Span pageLabel = new Span("Page");
041    protected MaterialListValueBox<Integer> listPages = new MaterialListValueBox<Integer>();
042
043    // Rows per Page Panel
044    protected MaterialColumn rowsPerPagePanel = new MaterialColumn();
045    protected Span rowsPerPageLabel = new Span("Rows per page");
046    protected MaterialListValueBox<Integer> listLimitOptions = new MaterialListValueBox<Integer>();
047
048    // Action Page Panel
049    protected MaterialColumn actionPagePanel = new MaterialColumn();
050    protected Span actionLabel = new Span();
051    protected MaterialIcon iconNext = new MaterialIcon(IconType.KEYBOARD_ARROW_RIGHT);
052    protected MaterialIcon iconPrev = new MaterialIcon(IconType.KEYBOARD_ARROW_LEFT);
053
054    public MaterialDataPagerBase() {
055        super(Document.get().createDivElement(), TableCssName.DATA_PAGER, TableCssName.ROW);
056    }
057
058    /**
059     * Build the number page panel - where user can change the page number into a listbox
060     */
061    protected void buildNumPagePanel() {
062        numPagePanel.setInitialClasses(TableCssName.NUM_PAGE_PANEL);
063        numPagePanel.setGrid("s12 m4 l3");
064        numPagePanel.setOffset("l3");
065        numPagePanel.add(listPages);
066        numPagePanel.add(pageLabel);
067        add(numPagePanel);
068    }
069
070    /**
071     * Build the rows per page panel - where you can set the range of row count into listbox
072     */
073    protected void buildLimitOptionsPanel() {
074        rowsPerPagePanel.setInitialClasses(TableCssName.ROWS_PER_PAGE_PANEL);
075        rowsPerPagePanel.setGrid("s12 m4 l3");
076        rowsPerPagePanel.add(listLimitOptions);
077        rowsPerPagePanel.add(rowsPerPageLabel);
078        add(rowsPerPagePanel);
079    }
080
081    /**
082     * Build the action panel - contains a page detail including the arrow next / previous icons.
083     */
084    protected void buildActionPanel() {
085        actionPagePanel.setInitialClasses(TableCssName.ACTION_PAGE_PANEL);
086        actionPagePanel.setGrid("s12 m4 l3");
087        actionLabel.setText("41 - 640 of 2014");
088        actionPagePanel.add(iconNext);
089        iconNext.setWaves(WavesType.DEFAULT);
090        iconNext.setCircle(true);
091        actionPagePanel.add(iconPrev);
092        iconPrev.setWaves(WavesType.DEFAULT);
093        iconPrev.setCircle(true);
094        actionPagePanel.add(actionLabel);
095        add(actionPagePanel);
096    }
097}
