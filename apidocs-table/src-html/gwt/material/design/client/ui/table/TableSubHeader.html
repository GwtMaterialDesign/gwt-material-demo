020package gwt.material.design.client.ui.table;
021
022import com.google.gwt.user.client.ui.Widget;
023import gwt.material.design.client.base.MaterialWidget;
024import gwt.material.design.client.base.constants.TableCssName;
025import gwt.material.design.client.constants.IconType;
026import gwt.material.design.client.data.component.CategoryComponent;
027import gwt.material.design.client.js.Js;
028import gwt.material.design.client.ui.MaterialIcon;
029import gwt.material.design.client.ui.html.Text;
030
031import java.util.NoSuchElementException;
032
033/**
034 * @author Ben Dol
035 */
036public class TableSubHeader extends TableRow {
037
038    private TableHeader iconTh;
039    private TableHeader nameTh;
040
041    private MaterialIcon icon;
042    private Text nameLbl;
043
044    private IconType openIcon = IconType.ADD;
045    private IconType closeIcon = IconType.REMOVE;
046
047    public TableSubHeader(TableSubHeader clone) {
048        this(clone.getDataCategory());
049        copy(clone);
050    }
051
052    public TableSubHeader(CategoryComponent category) {
053        super(category);
054        build();
055    }
056
057    @Override
058    protected void build() {
059        addStyleName(TableCssName.SUBHEADER);
060
061        iconTh = new TableHeader();
062        icon = new MaterialIcon(openIcon);
063        iconTh.add(icon);
064        add(iconTh);
065
066        nameTh = new TableHeader();
067        nameLbl = new Text("Subheader");
068        nameTh.add(nameLbl);
069        add(nameTh);
070
071        setName(category.getName());
072        setId(category.getName());
073    }
074
075    public void add(TableHeader tableHeader) {
076        super.add(tableHeader);
077    }
078
079    public void setName(String name) {
080        nameLbl.setText(name);
081    }
082
083    public String getName() {
084        return nameLbl.getText();
085    }
086
087    public void hideName() {
088        nameTh.setVisible(false);
089    }
090
091    public void showName() {
092        nameTh.setVisible(true);
093    }
094
095    public TableData getNameTh() {
096        return nameTh;
097    }
098
099    public IconType getOpenIcon() {
100        return openIcon;
101    }
102
103    public void setOpenIcon(IconType openIcon) {
104        this.openIcon = openIcon;
105
106        if(!isOpen()) {
107            icon.setIconType(openIcon);
108        }
109        setDataAttribute("data-open-icon", openIcon.getCssName());
110    }
111
112    public IconType getCloseIcon() {
113        return closeIcon;
114    }
115
116    public void setCloseIcon(IconType closeIcon) {
117        this.closeIcon = closeIcon;
118
119        if(isOpen()) {
120            icon.setIconType(closeIcon);
121        }
122        setDataAttribute("data-close-icon", closeIcon.getCssName());
123    }
124
125    public void hideIcon() {
126        iconTh.setVisible(false);
127    }
128
129    public void showIcon() {
130        iconTh.setVisible(true);
131    }
132
133    public TableData getIconTh() {
134        return iconTh;
135    }
136
137    public MaterialIcon getIcon() {
138        return icon;
139    }
140
141    public boolean isOpen() {
142        return $this().hasClass(TableCssName.EXPANDED);
143    }
144
145    public int getScrollPosition() {
146        int pos = Integer.valueOf(String.valueOf($this().data("pos")));
147        return !Js.isUndefinedOrNull(pos) ? pos : -1;
148    }
149
150    @Override
151    public void setVisible(boolean visible) {
152        super.setVisible(visible);
153        if(visible) {
154            $this().children("*").show();
155        } else {
156            $this().children("*").hide();
157        }
158    }
159
160    public void copy(TableSubHeader tableRow) {
161        if(tableRow != null) {
162            super.copy(tableRow);
163
164            icon.setIconType(tableRow.icon.getIconType());
165            nameLbl.setText(tableRow.nameLbl.getText());
166
167            setName(tableRow.getName());
168            setId(tableRow.getId());
169
170            openIcon = tableRow.openIcon;
171            closeIcon = tableRow.closeIcon;
172        }
173    }
174
175    @Override
176    public void removeFromParent() {
177        Widget parent = getParent();
178        if(parent != null && parent instanceof MaterialWidget) {
179            try {
180                ((MaterialWidget) parent).getChildren().remove(this);
181            } catch (NoSuchElementException ex) {
182                // This means we have already removed this subheader.
183            }
184        }
185        $this().parent().remove();
186    }
187
188    @Override
189    public String toString() {
190        return getElement().toString();
191    }
192}
