020package gwt.material.design.addins.client.avatar;
021
022import com.google.gwt.dom.client.Document;
023import gwt.material.design.addins.client.MaterialAddins;
024import gwt.material.design.addins.client.avatar.js.JsAvatar;
025import gwt.material.design.client.MaterialDesignBase;
026import gwt.material.design.client.base.MaterialWidget;
027
028//@formatter:off
029
030/**
031 * Generated avatar based on @link(https://jdenticon.com/)
032 * provides a unique avatar based on unique name.
033 * <p>
034 * <h3>XML Namespace Declaration</h3>
035 * <pre>
036 * {@code
037 * xmlns:ma='urn:import:gwt.material.design.addins.client'
038 * }
039 * </pre>
040 * <p>
041 * <h3>UiBinder Usage:</h3>
042 * <pre>
043 * {@code
044 *
045 * <ma:avatar.MaterialAvatar name="kevzlou7979" width="80" height="80"/>
046 *
047 * }
048 * </pre>
049 *
050 * @author kevzlou7979
051 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#avatar">Material Avatar</a>
052 */
053//@formatter:on
054public class MaterialAvatar extends MaterialWidget {
055
056    static {
057        if (MaterialAddins.isDebug()) {
058            MaterialDesignBase.injectDebugJs(MaterialAvatarDebugClientBundle.INSTANCE.jdenticonDebugJs());
059            MaterialDesignBase.injectDebugJs(MaterialAvatarDebugClientBundle.INSTANCE.md5DebugJs());
060        } else {
061            MaterialDesignBase.injectJs(MaterialAvatarClientBundle.INSTANCE.jdenticonJs());
062            MaterialDesignBase.injectJs(MaterialAvatarClientBundle.INSTANCE.md5Js());
063        }
064    }
065
066    private String name;
067
068    public MaterialAvatar() {
069        super(Document.get().createCanvasElement());
070    }
071
072    public MaterialAvatar(String name) {
073        this();
074        setName(name);
075    }
076
077    @Override
078    protected void initialize() {
079        if (getName() != null) {
080            getElement().setAttribute("data-jdenticon-hash", generateHashCode(getName()));
081            JsAvatar.jdenticon();
082        }
083    }
084
085    @Override
086    public void reinitialize() {
087        initialize();
088    }
089
090    /**
091     * Get the name of the avatar.
092     */
093    public String getName() {
094        return name;
095    }
096
097    /**
098     * Set the name of the avatar and hashed it using md5 js library to
099     * pass it into jdenticon avatar process.
100     */
101    public void setName(String name) {
102        this.name = name;
103    }
104
105    @Override
106    public void setWidth(String width) {
107        getElement().setAttribute("width", width);
108    }
109
110    @Override
111    public void setHeight(String height) {
112        getElement().setAttribute("height", height);
113    }
114
115    /**
116     * Generate hash code - needed by jdenticon to generate avatar.
117     */
118    protected String generateHashCode(String value) {
119        return JsAvatar.md5(value);
120    }
121}
