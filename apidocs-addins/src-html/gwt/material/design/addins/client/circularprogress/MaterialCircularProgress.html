021
022import com.google.gwt.core.client.Scheduler;
023import com.google.gwt.dom.client.Document;
024import gwt.material.design.addins.client.MaterialAddins;
025import gwt.material.design.addins.client.base.constants.AddinsCssName;
026import gwt.material.design.addins.client.circularprogress.events.*;
027import gwt.material.design.addins.client.circularprogress.js.JsCircularProgressOptions;
028import gwt.material.design.addins.client.circularprogress.ui.CircularProgressLabel;
029import gwt.material.design.client.MaterialDesignBase;
030import gwt.material.design.client.base.AbstractValueWidget;
031import gwt.material.design.client.base.JsLoader;
032import gwt.material.design.client.base.MaterialWidget;
033import gwt.material.design.client.base.helper.ColorHelper;
034import gwt.material.design.client.base.mixin.FontSizeMixin;
035import gwt.material.design.client.base.mixin.ToggleStyleMixin;
036import gwt.material.design.client.constants.Color;
037
038import static gwt.material.design.addins.client.circularprogress.js.JsCircularProgress.$;
039
040/**
041 * Circular Progress that draws an animated circular progress bar. Good for dashboards.
042 * <p>
043 * <h3>XML Namespace Declaration</h3>
044 * <pre>
045 * {@code
046 * xmlns:ma='urn:import:gwt.material.design.addins.client'
047 * }
048 * </pre>
049 * <p>
050 * <h3>UiBinder Usage:</h3>
051 * <pre>
052 * {@code
053 * <!-- Default -->
054 * <ma:circularprogress.MaterialCircularProgress value="0.3" grid="s12 m4 l3"/>
055 * <!-- Fill Color -->
056 * <ma:circularprogress.MaterialCircularProgress fillColor="PURPLE" emptyFillColor="PURPLE_LIGHTEN_4" value="0.3" grid="s12 m4 l3"/>
057 * <!-- Thickness -->
058 * <ma:circularprogress.MaterialCircularProgress value="0.3" thickness="20" grid="s12 m4 l3"/>
059 * }
060 * </pre>
061 *
062 * @author kevzlou7979
063 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#circularprogress">Material Circular Progress</a>
064 * @see <a href="https://github.com/kottenator/jquery-circle-progress">jquery-circle-progress 1.2.2</a>
065 */
066public class MaterialCircularProgress extends AbstractValueWidget<Double> implements JsLoader, HasCircularProgressHandlers {
067
068    static {
069        if (MaterialAddins.isDebug()) {
070            MaterialDesignBase.injectDebugJs(MaterialCircularProgressDebugClientBundle.INSTANCE.circularProgressDebugJs());
071            MaterialDesignBase.injectCss(MaterialCircularProgressDebugClientBundle.INSTANCE.circularProgressDebugCss());
072        } else {
073            MaterialDesignBase.injectJs(MaterialCircularProgressClientBundle.INSTANCE.circularProgressJs());
074            MaterialDesignBase.injectCss(MaterialCircularProgressClientBundle.INSTANCE.circularProgressCss());
075        }
076    }
077
078
079    private CircularProgressLabel label = new CircularProgressLabel();
080    private Color fillColor = Color.BLUE;
081    private Color emptyFillColor = Color.GREY_LIGHTEN_2;
082
083    private ToggleStyleMixin<MaterialWidget> responsiveMixin;
084    private JsCircularProgressOptions options = JsCircularProgressOptions.create();
085    private FontSizeMixin<MaterialWidget> fontSizeMixin;
086
087    public MaterialCircularProgress() {
088        super(Document.get().createDivElement(), AddinsCssName.CIRCULAR_PROGRESS);
089    }
090
091    @Override
092    protected void onLoad() {
093        super.onLoad();
094
095        label.setSize(getSize());
096        add(label);
097
098
099        $(getElement()).on(CircularProgressEvents.PROGRESS, (e, progress, step) -> {
100            ProgressEvent.fire(this, (double) progress, (double) step);
101            return true;
102        });
103        $(getElement()).on(CircularProgressEvents.START, (e, param1) -> {
104            StartEvent.fire(this, getValue());
105            return true;
106        });
107
108        $(getElement()).on(CircularProgressEvents.COMPLETED, (e, param1) -> {
109            CompleteEvent.fire(this, getValue());
110            return true;
111        });
112
113        load();
114    }
115
116    @Override
117    public void load() {
118        Scheduler.get().scheduleDeferred(() -> $(getElement()).circleProgress(options));
119    }
120
121    @Override
122    protected void onUnload() {
123        super.onUnload();
124
125        unload();
126    }
127
128    @Override
129    public void unload() {
130        $(getElement()).off(CircularProgressEvents.START);
131        $(getElement()).off(CircularProgressEvents.PROGRESS);
132        $(getElement()).off(CircularProgressEvents.COMPLETED);
133    }
134
135    @Override
136    public void reload() {
137        unload();
138        load();
139    }
140
141    @Override
142    public Double getValue() {
143        return options.value;
144    }
145
146    @Override
147    public void setValue(Double value) {
148        setValue(value, false);
149        options.value = value;
150    }
151
152    /**
153     * This is the only required option. It should be from 0.0 to 1.0 (Default: 0)
154     */
155    public Color getFillColor() {
156        return fillColor;
157    }
158
159    /**
160     * Set the fillColor of the circular progress
161     */
162    public void setFillColor(Color fillColor) {
163        this.fillColor = fillColor;
164        options.fill = ColorHelper.setupComputedBackgroundColor(fillColor);
165    }
166
167    public Color getEmptyFillColor() {
168        return emptyFillColor;
169    }
170
171    /**
172     * Set the empty fill color of the circular progress
173     */
174    public void setEmptyFillColor(Color emptyFillColor) {
175        this.emptyFillColor = emptyFillColor;
176        options.emptyFill = ColorHelper.setupComputedBackgroundColor(emptyFillColor);
177    }
178
179    public double getSize() {
180        return options.size;
181    }
182
183    /**
184     * Set the size of the circle / canvas in pixels
185     */
186    public void setSize(int size) {
187        options.size = size;
188    }
189
190    public int getThickness() {
191        return (int) options.thickness;
192    }
193
194    /**
195     * Thickness width of the arc. (Default : 8)
196     */
197    public void setThickness(int thickness) {
198        options.thickness = thickness;
199    }
200
201    public double getStartAngle() {
202        return options.startAngle;
203    }
204
205    /**
206     * Initial angle of the circular progress component (Default : -Math.PI);
207     */
208    public void setStartAngle(double startAngle) {
209        options.startAngle = startAngle;
210    }
211
212    public boolean isReverse() {
213        return options.reverse;
214    }
215
216    /**
217     * Reverse animation and arc draw (Default : false)
218     */
219    public void setReverse(boolean reverse) {
220        options.reverse = reverse;
221    }
222
223    /**
224     * Set the text of the circular progress
225     */
226    public void setText(String text) {
227        label.setText(text);
228    }
229
230    public boolean isResponsive() {
231        return getResponsiveMixin().isOn();
232    }
233
234    /**
235     * Enable Responsive circular progress
236     * If responsive we will set the size to maximum value = 1000,
237     * else set it to the default size = 100
238     **/
239    public void setResponsive(boolean responsive) {
240        options.size = responsive == true ? 1000 : 100;
241        getResponsiveMixin().setOn(responsive);
242    }
243
244    public CircularProgressLabel getLabel() {
245        return label;
246    }
247
248    public ToggleStyleMixin<MaterialWidget> getResponsiveMixin() {
249        if (responsiveMixin == null) {
250            responsiveMixin = new ToggleStyleMixin(this, AddinsCssName.RESPONSIVE);
251        }
252        return responsiveMixin;
253    }
254
255    @Override
256    protected FontSizeMixin<MaterialWidget> getFontSizeMixin() {
257        if (fontSizeMixin == null) {
258            fontSizeMixin = new FontSizeMixin<>(label);
259        }
260        return fontSizeMixin;
261    }
262
263    @Override
264    public void addProgressHandler(ProgressEvent.ProgressHandler handler) {
265        addHandler(handler, ProgressEvent.TYPE);
266    }
267
268    @Override
269    public void addStartHandler(StartEvent.StartHandler handler) {
270        addHandler(handler, StartEvent.TYPE);
271    }
272
273    @Override
274    public void addCompleteHandler(CompleteEvent.CompleteHandler handler) {
275        addHandler(handler, CompleteEvent.TYPE);
276    }
277}
