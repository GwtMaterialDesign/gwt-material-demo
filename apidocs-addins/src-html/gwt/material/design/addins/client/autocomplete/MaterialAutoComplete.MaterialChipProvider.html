048/**
049 * <p>
050 * Use GWT Autocomplete to search for matches from local or remote data sources.
051 * We used MultiWordSuggestOracle to populate the list to be added on the
052 * autocomplete values.
053 * </p>
054 *
055 * <h3>XML Namespace Declaration</h3>
056 * <pre>
057 * {@code
058 * xmlns:ma='urn:import:gwt.material.design.addins.client'
059 * }
060 * </pre>
061 *
062 * <h3>UiBinder Usage:</h3>
063 * <pre>
064 * {@code
065 *  <ma:autocomplete.MaterialAutoComplete ui:field="autocomplete" placeholder="States" />}
066 * </pre>
067 *
068 * <h3>Java Usage:</h3>
069 *
070 * <p>
071 * To use your domain object inside the MaterialAutoComplete, for example, an object
072 * "User", you can subclass the {@link gwt.material.design.addins.client.autocomplete.base.MaterialSuggestionOracle} and {@link Suggestion}, like this:
073 * </p>
074 * <p><pre>
075 * public class UserOracle extends MaterialSuggestionOracle {
076 *     private List&lt;User&gt; contacts = new LinkedList&lt;&gt;();
077 *
078 *     public void addContacts(List&lt;User&gt; users){
079 *         contacts.addAll(users);
080 *     }
081 *
082 *     {@literal @}Override
083 *     public void requestSuggestions(final Request request, final Callback callback) {
084 *         Response resp = new Response();
085 *         if (contacts.isEmpty()){
086 *             callback.onSuggestionsReady(request, resp);
087 *             return;
088 *         }
089 *         String text = request.getQuery();
090 *         text = text.toLowerCase();
091 *
092 *         List&lt;UserSuggestion&gt; list = new ArrayList&lt;&gt;();
093 *
094 *         /{@literal *}
095 *         {@literal *}  Finds the contacts that meets the criteria. Note that since the 
096 *         {@literal *}  requestSuggestions method is asynchronous, you can fetch the
097 *         {@literal *}  results from the server instead of using a local contacts List.
098 *         {@literal *}/
099 *         for (User contact : contacts){
100 *             if (contact.getName().toLowerCase().contains(text)){
101 *                 list.add(new UserSuggestion(contact));
102 *             }
103 *          }
104 *          resp.setSuggestions(list);
105 *          callback.onSuggestionsReady(request, resp);
106 *     }
107 * }
108 *
109 * public class UserSuggestion implements SuggestOracle.Suggestion {
110 *
111 *     private User user;
112 *
113 *     public UserSuggestion(User user){
114 *         this.user = user;
115 *     }
116 *
117 *     {@literal @}Override
118 *     public String getDisplayString() {
119 *         return getReplacementString();
120 *     }
121 *
122 *     {@literal @}Override
123 *     public String getReplacementString() {
124 *         return user.getName();
125 *     }
126 *
127 *     public User getUser() {
128 *         return user;
129 *     }
130 * }
131 * </pre></p>
132 * <p>
133 * And then use the UserOracle like this:
134 * </p>
135 * <p><pre>
136 * //Constructor
137 * MaterialAutoComplete userAutoComplete = new MaterialAutoComplete(new UserOracle());
138 *
139 * //How to get the selected User objects
140 * public List&lt;User&gt; getSelectedUsers(){
141 *     List&lt;? extends Suggestion&gt; values = userAutoComplete.getValue();
142 *     List&lt;User&gt; users = new ArrayList&lt;&gt;(values.size());
143 *     for (Suggestion value : values) {
144 *         if (value instanceof UserSuggestion){
145 *             UserSuggestion us = (UserSuggestion) value;
146 *             User user = us.getUser();                            
147 *             users.add(user);
148 *         }
149 *     }
150 *     return users;
151 * }
152 * </pre></p>
153 *
154 * @author kevzlou7979
155 * @author gilberto-torrezan
156 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#!autocomplete">Material AutoComplete</a>
157 */

215    /**
216     * Generate and build the List Items to be set on Auto Complete box.
217     */
218    private void generateAutoComplete(SuggestOracle suggestions) {
219        list.setStyleName("multiValueSuggestBox-list");
220        this.suggestions = suggestions;
221        final ListItem item = new ListItem();
222
223        item.setStyleName("multiValueSuggestBox-input-token");
224        box = new SuggestBox(suggestions, itemBox);
225        setLimit(this.limit);
226        String autocompleteId = DOM.createUniqueId();
227        itemBox.getElement().setId(autocompleteId);
228
229        item.add(box);
230        list.add(item);
231
232        list.addDomHandler(new ClickHandler() {
233            @Override
234            public void onClick(ClickEvent event) {
235                box.showSuggestionList();
236            }
237        }, ClickEvent.getType());
238
239        itemBox.addKeyDownHandler(new KeyDownHandler() {
240            public void onKeyDown(KeyDownEvent event) {
241                boolean itemsChanged = false;
242
243                switch (event.getNativeKeyCode()) {
244                    case KeyCodes.KEY_ENTER:
245                        if (directInputAllowed) {
246                            String value = itemBox.getValue();
247                            if (value != null && !(value = value.trim()).isEmpty()) {
248                                gwt.material.design.client.base.Suggestion directInput = new gwt.material.design.client.base.Suggestion();
249                                directInput.setDisplay(value);
250                                directInput.setSuggestion(value);
251                                itemsChanged = addItem(directInput);
252                                itemBox.setValue("");
253                                itemBox.setFocus(true);
254                            }
255                        }
256                        break;
257
258                    case KeyCodes.KEY_BACKSPACE:
259                        if (itemBox.getValue().trim().isEmpty()) {
260                            if (itemsHighlighted.isEmpty()) {
261                                if (suggestionMap.size() > 0) {
262
263                                    ListItem li = (ListItem) list.getWidget(list.getWidgetCount() - 2);
264                                    MaterialChip p = (MaterialChip) li.getWidget(0);
265                                    
266                                    boolean removable = true;
267                                    
268                                    Set<Entry<Suggestion, Widget>> entrySet = suggestionMap.entrySet();
269                                    for (Entry<Suggestion, Widget> entry : entrySet) {
270                                        if (p.equals(entry.getValue())) {
271                                            if (chipProvider.isChipRemovable(entry.getKey())){
272                                                suggestionMap.remove(entry.getKey());
273                                                itemsChanged = true;                                                
274                                            }
275                                            else {
276                                                removable = false;
277                                            }
278                                            break;
279                                        }
280                                    }
281
282                                    if (removable){
283                                        list.remove(li);                                        
284                                    }
285                                }
286                            }
287                        }
288
289                    case KeyCodes.KEY_DELETE:
290                        if (itemBox.getValue().trim().isEmpty()) {
291                            for (ListItem li : itemsHighlighted) {
292                                MaterialChip p = (MaterialChip) li.getWidget(0);
293                                
294                                boolean removable = true;
295                                
296                                Set<Entry<Suggestion, Widget>> entrySet = suggestionMap.entrySet();
297                                for (Entry<Suggestion, Widget> entry : entrySet) {
298                                    if (p.equals(entry.getValue())) {
299                                        if (chipProvider.isChipRemovable(entry.getKey())){
300                                            suggestionMap.remove(entry.getKey());
301                                            itemsChanged = true;                                            
302                                        }
303                                        else {
304                                            removable = false;
305                                        }
306                                        break;
307                                    }
308                                }
309                                
310                                if (removable){
311                                    li.removeFromParent();                                   
312                                }
313                            }
314                            itemsHighlighted.clear();
315                        }
316                        itemBox.setFocus(true);
317                        break;
318                }
319
320                if (itemsChanged) {
321                    ValueChangeEvent.fire(MaterialAutoComplete.this, getValue());
322                }
323            }
324        });
325
326        itemBox.addClickHandler(new ClickHandler() {
327            @Override
328            public void onClick(ClickEvent event) {
329                box.showSuggestionList();
330            }
331        });
332
333        box.addSelectionHandler(new SelectionHandler<Suggestion>() {
334            public void onSelection(SelectionEvent<Suggestion> selectionEvent) {
335                Suggestion selectedItem = selectionEvent.getSelectedItem();
336                itemBox.setValue("");
337                if (addItem(selectedItem)) {
338                    ValueChangeEvent.fire(MaterialAutoComplete.this, getValue());
339                }
340                itemBox.setFocus(true);
341            }
342        });
343
344        panel.add(list);
345        panel.getElement().setAttribute("onclick",
346                "document.getElementById('" + autocompleteId + "').focus()");
347        panel.add(lblError);
348        box.setFocus(true);
349    }

351    /**
352     * Adding the item value using Material Chips added on auto complete box
353     */
354    protected boolean addItem(final Suggestion suggestion) {
355        SelectionEvent.fire(MaterialAutoComplete.this, suggestion);
356        if (getLimit() > 0) {
357            if (suggestionMap.size() >= getLimit()) {
358                return false;
359            }
360        }
361
362        if (suggestionMap.containsKey(suggestion)) {
363            return false;
364        }
365
366        final ListItem displayItem = new ListItem();
367        displayItem.setStyleName("multiValueSuggestBox-token");
368
369        if (getType() == AutocompleteType.TEXT){
370            suggestionMap.clear();
371            itemBox.setText(suggestion.getDisplayString());
372            displayItem.add(itemBox);
373        }
374        else {
375            final MaterialChip chip = chipProvider.getChip(suggestion);
376            if (chip == null) {
377                return false;
378            }
379
380            chip.addClickHandler(new ClickHandler() {
381                public void onClick(ClickEvent clickEvent) {
382                    if (chipProvider.isChipSelectable(suggestion)){
383                        if (itemsHighlighted.contains(displayItem)) {
384                            chip.removeStyleName(selectedChipStyle);
385                            itemsHighlighted.remove(displayItem);
386                        } else {
387                            chip.addStyleName(selectedChipStyle);
388                            itemsHighlighted.add(displayItem);
389                        }
390                    }
391                }
392            });                
393            
394            if (chip.getIcon() != null){
395                chip.getIcon().addClickHandler(new ClickHandler() {
396                    public void onClick(ClickEvent clickEvent) {
397                        if (chipProvider.isChipRemovable(suggestion)){
398                            suggestionMap.remove(suggestion);
399                            list.remove(displayItem);
400                            itemsHighlighted.remove(displayItem);
401                            ValueChangeEvent.fire(MaterialAutoComplete.this, getValue());
402                            box.showSuggestionList();
403                        }
404                    }
405                });                
406            }
407
408            suggestionMap.put(suggestion, chip);
409            displayItem.add(chip);
410        }
411
412        list.insert(displayItem, list.getWidgetCount() - 1);
413        return true;
414    }

416    /**
417     * Clear the chip items on the autocomplete box
418     */
419    public void clear() {
420        itemBox.setValue("");
421
422        Collection<Widget> values = suggestionMap.values();
423        for (Widget widget : values) {
424            Widget parent = widget.getParent();
425            if (parent instanceof ListItem) {
426                parent.removeFromParent();
427            }
428        }
429        suggestionMap.clear();
430
431        clearErrorOrSuccess();
432    }

453        if (itemValues == null) {
454            clear();
455            return;
456        }
457        List<Suggestion> list = new ArrayList<>(itemValues.size());
458        for (String value : itemValues) {
459            Suggestion suggestion = new gwt.material.design.client.base.Suggestion(value, value);
460            list.add(suggestion);
461        }
462        setValue(list);

693    /**
694     * Default implementation of the {@link MaterialChipProvider} interface,
695     * used by the {@link gwt.material.design.addins.client.autocomplete.MaterialAutoComplete}.
696     * 
697     * <p>
698     * By default all chips are selectable and removable. The default {@link IconType} used by the chips provided is the {@link IconType#CLOSE}.
699     * </p> 
700     *
701     * @see gwt.material.design.addins.client.autocomplete.MaterialAutoComplete#setChipProvider(MaterialChipProvider)
702     */
703    public static class DefaultMaterialChipProvider implements MaterialChipProvider {
704
705        @Override
706        public MaterialChip getChip(Suggestion suggestion) {
707            final MaterialChip chip = new MaterialChip();
708
709            String imageChip = suggestion.getDisplayString();
710            String textChip = imageChip;
711
712            String s = "<img src=\"";
713            if (imageChip.contains(s)) {
714                int ix = imageChip.indexOf(s) + s.length();
715                imageChip = imageChip.substring(ix, imageChip.indexOf("\"", ix + 1));
716                chip.setUrl(imageChip);
717                textChip = textChip.replaceAll("[<](/)?img[^>]*[>]", "");
718            }
719            chip.setText(textChip);
720            chip.setIconType(IconType.CLOSE);
721            
722            return chip;
723        }
724        
725        @Override
726        public boolean isChipRemovable(Suggestion suggestion) {
727            return true;
728        }
729        
730        @Override
731        public boolean isChipSelectable(Suggestion suggestion) {
732            return true;
733        }
734    }
