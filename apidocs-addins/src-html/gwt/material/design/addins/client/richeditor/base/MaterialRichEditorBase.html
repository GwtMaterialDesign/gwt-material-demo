020package gwt.material.design.addins.client.richeditor.base;
021
022import com.google.gwt.core.client.JsArrayString;
023import com.google.gwt.dom.client.Document;
024import com.google.gwt.dom.client.Element;
025import com.google.gwt.event.shared.HandlerRegistration;
026import com.google.gwt.safehtml.shared.SafeHtmlUtils;
027import com.google.gwt.user.client.ui.HasHTML;
028import gwt.material.design.addins.client.base.constants.AddinsCssName;
029import gwt.material.design.addins.client.richeditor.base.constants.ToolbarButton;
030import gwt.material.design.client.base.AbstractValueWidget;
031import gwt.material.design.client.base.HasPlaceholder;
032
033import static gwt.material.design.addins.client.richeditor.js.JsRichEditor.$;
034
035/**
036 * Base widget for Rich Editor with specific properties for options
037 */
038public class MaterialRichEditorBase extends AbstractValueWidget<String> implements HasPlaceholder, HasHTML {
039
040    private boolean airMode;
041    private String placeholder = "";
042    private boolean disableDragAndDrop;
043
044    public MaterialRichEditorBase() {
045        super(Document.get().createDivElement(), AddinsCssName.EDITOR);
046    }
047
048    private ToolbarButton[] styleOptions = new ToolbarButton[]
049            {ToolbarButton.STYLE, ToolbarButton.BOLD, ToolbarButton.ITALIC, ToolbarButton.UNDERLINE, ToolbarButton.STRIKETHROUGH, ToolbarButton.CLEAR, ToolbarButton.SUPERSCRIPT, ToolbarButton.SUBSCRIPT};
050    private ToolbarButton[] fontOptions = new ToolbarButton[]
051            {ToolbarButton.FONT_SIZE, ToolbarButton.FONT_NAME};
052    private ToolbarButton[] colorOptions = new ToolbarButton[]
053            {ToolbarButton.COLOR};
054    private ToolbarButton[] undoOptions = new ToolbarButton[]
055            {ToolbarButton.UNDO, ToolbarButton.REDO, ToolbarButton.HELP};
056    private ToolbarButton[] ckMediaOptions = new ToolbarButton[]
057            {ToolbarButton.CK_IMAGE_UPLOAD, ToolbarButton.CK_IMAGE_VIDEO};
058    private ToolbarButton[] miscOptions = new ToolbarButton[]
059            {ToolbarButton.LINK, ToolbarButton.PICTURE, ToolbarButton.TABLE, ToolbarButton.HR, ToolbarButton.CODE_VIEW, ToolbarButton.FULLSCREEN};
060    private ToolbarButton[] paraOptions = new ToolbarButton[]
061            {ToolbarButton.UL, ToolbarButton.OL, ToolbarButton.PARAGRAPH, ToolbarButton.LEFT, ToolbarButton.CENTER, ToolbarButton.RIGHT, ToolbarButton.JUSTIFY, ToolbarButton.OUTDENT, ToolbarButton.INDENT};
062    private ToolbarButton[] heightOptions = new ToolbarButton[]
063            {ToolbarButton.LINE_HEIGHT};
064
065    private HandlerRegistration htmlAttachHandler;
066
067    public JsArrayString extractOptions(ToolbarButton[] options) {
068        JsArrayString jsOptions = JsArrayString.createArray().cast();
069        for (ToolbarButton option : options) {
070            jsOptions.push(option.getId());
071        }
072        return jsOptions;
073    }
074
075    public ToolbarButton[] getStyleOptions() {
076        return styleOptions;
077    }
078
079    public void setStyleOptions(ToolbarButton... styleOptions) {
080        this.styleOptions = styleOptions;
081    }
082
083    public ToolbarButton[] getFontOptions() {
084        return fontOptions;
085    }
086
087    public void setFontOptions(ToolbarButton... fontOptions) {
088        this.fontOptions = fontOptions;
089    }
090
091    public ToolbarButton[] getColorOptions() {
092        return colorOptions;
093    }
094
095    public void setColorOptions(ToolbarButton... colorOptions) {
096        this.colorOptions = colorOptions;
097    }
098
099    public ToolbarButton[] getUndoOptions() {
100        return undoOptions;
101    }
102
103    public void setUndoOptions(ToolbarButton... undoOptions) {
104        this.undoOptions = undoOptions;
105    }
106
107    public ToolbarButton[] getCkMediaOptions() {
108        return ckMediaOptions;
109    }
110
111    public void setCkMediaOptions(ToolbarButton... ckMediaOptions) {
112        this.ckMediaOptions = ckMediaOptions;
113    }
114
115    public ToolbarButton[] getMiscOptions() {
116        return miscOptions;
117    }
118
119    public void setMiscOptions(ToolbarButton... miscOptions) {
120        this.miscOptions = miscOptions;
121    }
122
123    public ToolbarButton[] getParaOptions() {
124        return paraOptions;
125    }
126
127    public void setParaOptions(ToolbarButton... paraOptions) {
128        this.paraOptions = paraOptions;
129    }
130
131    public ToolbarButton[] getHeightOptions() {
132        return heightOptions;
133    }
134
135    public void setHeightOptions(ToolbarButton... heightOptions) {
136        this.heightOptions = heightOptions;
137    }
138
139    public boolean isAirMode() {
140        return airMode;
141    }
142
143    public void setAirMode(boolean airMode) {
144        this.airMode = airMode;
145    }
146
147    @Override
148    public String getPlaceholder() {
149        return placeholder;
150    }
151
152    @Override
153    public void setPlaceholder(String placeholder) {
154        this.placeholder = placeholder;
155    }
156
157    public String getHeight() {
158        String height = getElement().getStyle().getHeight();
159        if (height == null || height.isEmpty()) {
160            height = "550px";
161        }
162        return height;
163    }
164
165    /**
166     * Check if the dnd for rich editor is enabled / disabled
167     */
168    public boolean isDisableDragAndDrop() {
169        return disableDragAndDrop;
170    }
171
172    /**
173     * If true, disable the ability to drag and drop items to rich editor
174     */
175    public void setDisableDragAndDrop(boolean disableDragAndDrop) {
176        this.disableDragAndDrop = disableDragAndDrop;
177    }
178
179    @Override
180    public String getHTML() {
181        return getHTMLCode(getElement());
182    }
183
184    @Override
185    public void setHTML(final String html) {
186        if (htmlAttachHandler != null) {
187            htmlAttachHandler.removeHandler();
188            htmlAttachHandler = null;
189        }
190
191        if (!isAttached()) {
192
193            htmlAttachHandler = addAttachHandler(e -> setHTMLCode(getElement(), html));
194        } else {
195
196            setHTMLCode(getElement(), html);
197        }
198    }
199
200    @Override
201    public String getText() {
202        return getElement().getInnerText();
203    }
204
205    @Override
206    public void setText(String text) {
207        getElement().setInnerSafeHtml(SafeHtmlUtils.fromString(text));
208    }
209
210    protected String getHTMLCode(Element e) {
211        return $(e).code();
212    }
213
214    protected void setHTMLCode(Element e, String html) {
215        $(e).code(html);
216    }
217
218    @Override
219    public String getValue() {
220        return getText();
221    }
222
223    @Override
224    public void setValue(String value, boolean fireEvents) {
225        setText(value);
226
227        // We won't invoke the super call since the internal
228        // library will handle change events.
229    }
230}
