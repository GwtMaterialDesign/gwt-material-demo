020package gwt.material.design.addins.client.masonry;
021
022import com.google.gwt.dom.client.Document;
023import com.google.gwt.dom.client.Element;
024import com.google.gwt.user.client.ui.IsWidget;
025import com.google.gwt.user.client.ui.Widget;
026import gwt.material.design.addins.client.MaterialAddins;
027import gwt.material.design.addins.client.base.constants.AddinsCssName;
028import gwt.material.design.addins.client.masonry.js.JsMasonryOptions;
029import gwt.material.design.client.MaterialDesignBase;
030import gwt.material.design.client.base.MaterialWidget;
031import gwt.material.design.client.constants.CssName;
032import gwt.material.design.client.ui.MaterialRow;
033
034import static gwt.material.design.addins.client.masonry.js.JsMasonry.$;
035
036//@formatter:off
037
038/**
039 * Masonry works by placing elements in optimal position based on available vertical space, sort of like a mason fitting stones in a wall.
040 * <p>
041 * <h3>XML Namespace Declaration</h3>
042 * <pre>
043 * {@code
044 * xmlns:m.addins='urn:import:gwt.material.design.addins.client.ui'
045 * }
046 * </pre>
047 * <p>
048 * <h3>UiBinder Usage:</h3>
049 * <pre>
050 * {
051 * @code
052 * <m.addins:MaterialMasonry>
053 *     <m:MaterialColumn grid="l1" padding="4" backgroundColor="blue" height="200px">
054 *         <m:MaterialLabel text="1"/>
055 *     </m:MaterialColumn>
056 *     &lt;-- Other columns here -->
057 * </m.addins:MaterialMasonry>
058 * }
059 * </pre>
060 *
061 * @author kevzlou7979
062 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#masonry">Material Masonry</a>
063 */
064//@formatter:on
065public class MaterialMasonry extends MaterialRow {
066
067    static {
068        if (MaterialAddins.isDebug()) {
069            MaterialDesignBase.injectDebugJs(MaterialMasonryDebugClientBundle.INSTANCE.masonryJsDebug());
070            MaterialDesignBase.injectDebugJs(MaterialMasonryDebugClientBundle.INSTANCE.imageLoadedJsDebug());
071        } else {
072            MaterialDesignBase.injectJs(MaterialMasonryClientBundle.INSTANCE.masonryJs());
073            MaterialDesignBase.injectJs(MaterialMasonryClientBundle.INSTANCE.imageLoadedJs());
074        }
075    }
076
077    private String itemSelector = ".col";
078    private boolean percentPosition = true;
079    private boolean originLeft = true;
080    private boolean originTop = true;
081    private double transitionDuration = 400;
082    private boolean initialize;
083
084    private MaterialWidget sizerDiv = new MaterialWidget(Document.get().createDivElement());
085
086    public MaterialMasonry() {
087        super(Document.get().createDivElement(), AddinsCssName.MASONRY, CssName.ROW);
088        enableFeature(Feature.ONLOAD_ADD_QUEUE, true);
089        sizerDiv.setWidth("8.3333%");
090        sizerDiv.setStyleName(AddinsCssName.COL_SIZER);
091        add(sizerDiv);
092    }
093
094    @Override
095    protected void onLoad() {
096        super.onLoad();
097        initMasonry();
098
099        if (!initialize) {
100            initialize = true;
101        }
102    }
103
104    public void initMasonry() {
105        initMasonryJs(getElement());
106    }
107
108    protected void initMasonryJs(Element e) {
109        window().ready(() -> $(e).imagesLoaded(() -> $(e).masonry(getMasonryOptions())));
110    }
111
112    protected JsMasonryOptions getMasonryOptions() {
113        JsMasonryOptions options = new JsMasonryOptions();
114        options.itemSelector = ".masonry > " + getItemSelector();
115        options.percentPosition = isPercentPosition();
116        options.columnWidth = "." + AddinsCssName.COL_SIZER;
117        options.originLeft = isOriginLeft();
118        options.originTop = isOriginTop();
119        options.transitionDuration = getTransitionDuration() + "ms";
120        return options;
121    }
122
123    @Override
124    public boolean remove(IsWidget child) {
125        Widget widget = (Widget) child;
126        remove(widget.getElement());
127
128        if (initialize) {
129            initMasonry();
130        }
131        return true;
132    }
133
134    @Override
135    public boolean remove(Widget w) {
136        return this.remove((IsWidget) w);
137    }
138
139    @Override
140    public boolean remove(int index) {
141        remove(getWidget(index).getElement());
142        return true;
143    }
144
145    /**
146     * Remove the item with Masonry support
147     */
148    protected void remove(Element e) {
149        if (initialize) {
150            $(getElement()).masonry(getMasonryOptions()).masonry("remove", e).masonry("layout");
151        }
152    }
153
154    @Override
155    public void clear() {
156        for (Widget widget : getChildren()) {
157            remove(widget.getElement());
158        }
159    }
160
161    @Override
162    public void add(Widget child) {
163        super.add(child);
164        reload();
165    }
166
167    @Override
168    protected void add(Widget child, com.google.gwt.user.client.Element container) {
169        super.add(child, container);
170        reload();
171    }
172
173    @Override
174    protected void insert(Widget child, com.google.gwt.user.client.Element container, int beforeIndex, boolean domInsert) {
175        super.insert(child, container, beforeIndex, domInsert);
176        reload();
177    }
178
179    @Override
180    public void insert(Widget child, int beforeIndex) {
181        super.insert(child, beforeIndex);
182        reload();
183    }
184
185    /**
186     * Reload the layout effective only when adding and inserting items
187     */
188    public void reload() {
189        if (initialize) {
190            reloadItems();
191            layout();
192        }
193    }
194
195    /**
196     * Reload all items inside the masonry
197     */
198    public void reloadItems() {
199        $(getElement()).masonry(getMasonryOptions()).masonry("reloadItems");
200    }
201
202    /**
203     * Layout remaining item elements
204     */
205    public void layout() {
206        $(getElement()).masonry(getMasonryOptions()).masonry("layout");
207    }
208
209    /**
210     * Get the item selector.
211     */
212    public String getItemSelector() {
213        return itemSelector;
214    }
215
216    /**
217     * Specifies which child elements will be used as item elements in the
218     * layout.It's .col by default for grid components.
219     */
220    public void setItemSelector(String itemSelector) {
221        this.itemSelector = itemSelector;
222    }
223
224    /**
225     * Get the percent position boolean value.
226     */
227    public boolean isPercentPosition() {
228        return percentPosition;
229    }
230
231    /**
232     * Sets item positions in percent values, rather than pixel values. percentPosition: true works
233     * well with percent-width items, as items will not transition their position on resize.
234     */
235    public void setPercentPosition(boolean percentPosition) {
236        this.percentPosition = percentPosition;
237    }
238
239    /**
240     * Get the boolean value of origin left.
241     */
242    public boolean isOriginLeft() {
243        return originLeft;
244    }
245
246    /**
247     * Controls the horizontal flow of the layout. By default, item elements start positioning at the
248     * left, with originLeft: true. Set originLeft: false for right-to-left layouts.
249     */
250    public void setOriginLeft(boolean originLeft) {
251        this.originLeft = originLeft;
252    }
253
254    /**
255     * Get the boolean value of origin top.
256     */
257    public boolean isOriginTop() {
258        return originTop;
259    }
260
261    /**
262     * Controls the vertical flow of the layout. By default, item elements start positioning at the top,
263     * with originTop: true. Set originTop: false for bottom-up layouts. Itâ€™s like Tetris
264     */
265    public void setOriginTop(boolean originTop) {
266        this.originTop = originTop;
267    }
268
269    /**
270     * Get the transition duration in milliseconds.
271     */
272    public double getTransitionDuration() {
273        return transitionDuration;
274    }
275
276    /**
277     * Sets the transition duration in milliseconds, if 0 then there will be no transition.
278     */
279    public void setTransitionDuration(double transitionDuration) {
280        this.transitionDuration = transitionDuration;
281    }
282
283    public MaterialWidget getSizerDiv() {
284        return sizerDiv;
285    }
286}
