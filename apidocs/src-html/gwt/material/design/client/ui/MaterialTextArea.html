023import gwt.material.design.client.base.ComplexWidget;
024import gwt.material.design.client.base.HasError;
025import gwt.material.design.client.base.HasIcon;
026import gwt.material.design.client.base.HasPlaceholder;
027import gwt.material.design.client.base.mixin.ErrorMixin;
028import gwt.material.design.client.constants.IconPosition;
029import gwt.material.design.client.constants.IconSize;
030import gwt.material.design.client.constants.IconType;
031import gwt.material.design.client.ui.html.Label;
032
033import com.google.gwt.dom.client.Document;
034import com.google.gwt.dom.client.Style;
035import com.google.gwt.event.dom.client.ChangeEvent;
036import com.google.gwt.event.dom.client.ChangeHandler;
037import com.google.gwt.event.dom.client.HasChangeHandlers;
038import com.google.gwt.event.dom.client.HasKeyDownHandlers;
039import com.google.gwt.event.dom.client.HasKeyPressHandlers;
040import com.google.gwt.event.dom.client.HasKeyUpHandlers;
041import com.google.gwt.event.dom.client.KeyDownEvent;
042import com.google.gwt.event.dom.client.KeyDownHandler;
043import com.google.gwt.event.dom.client.KeyPressEvent;
044import com.google.gwt.event.dom.client.KeyPressHandler;
045import com.google.gwt.event.dom.client.KeyUpEvent;
046import com.google.gwt.event.dom.client.KeyUpHandler;
047import com.google.gwt.event.shared.HandlerRegistration;
048import com.google.gwt.user.client.ui.HasText;

050//@formatter:off
051/**
052* Material Text Area represents a multiple line textbox where users can define comment, detail and etc.
053* <h3>UiBinder Usage:</h3>
054* <pre>
055*{@code <m:MaterialTextArea placeholder="Your Comment" />
056* </pre>
057* @see <a href="http://gwt-material-demo.herokuapp.com/#forms">Material TextArea</a>
058* @author kevzlou7979
059* @author Ben Dol
060*/
061//@formatter:on
062public class MaterialTextArea extends ComplexWidget implements HasText, HasKeyPressHandlers,
063        HasKeyDownHandlers, HasKeyUpHandlers, HasChangeHandlers, HasError, HasIcon, HasPlaceholder {
064
065    private String length;
066    private String placeholder;
067    private boolean isValid = true;
068
069    private MaterialLabel lblError = new MaterialLabel();
070
071    private final ErrorMixin<MaterialTextArea, MaterialLabel> errorMixin = new ErrorMixin<>(this, lblError, null);
072
073    private Label label = new Label();
074    private MaterialLabel lblName = new MaterialLabel();
075    private ComplexWidget textArea = new ComplexWidget(Document.get().createElement("textarea"));
076    private MaterialIcon icon = new MaterialIcon();
077
078    public MaterialTextArea() {
079        super(Document.get().createDivElement());
080        setStyleName("input-field");
081        add(icon);
082        add(textArea);
083        textArea.setStyleName("materialize-textarea");
084        add(label);
085        label.add(lblName);
086        lblError.setVisible(false);
087        add(lblError);
088    }
089
090    public void setInvalid() {
091        removeErrorModifiers();
092        lblName.setStyleName("red-text");
093        textArea.getElement().addClassName("invalid");
094        isValid = false;
095    }
096
097    public void setValid() {
098        removeErrorModifiers();
099        lblName.setStyleName("green-text");
100        textArea.getElement().addClassName("valid");
101        isValid = true;
102    }
103
104    public void removeErrorModifiers() {
105        textArea.getElement().removeClassName("valid");
106        textArea.getElement().removeClassName("invalid");
107    }
108
109    public boolean isValid() {
110        return isValid;
111    }
112
113    public void setValid(boolean isValid) {
114        this.isValid = isValid;
115    }
116
117    public String getLength() {
118        return length;
119    }
120
121    public void setLength(String length) {
122        this.length = length;
123        textArea.getElement().setAttribute("length", length);
124    }
125
126    @Override
127    protected void onLoad() {
128        super.onLoad();
129        label.getElement().setAttribute("for", "field");
130    }
131
132    @Override
133    public String getText() {
134        return textArea.getElement().getInnerHTML();
135    }
136
137    @Override
138    public void setText(String text) {
139        textArea.getElement().setInnerHTML(text);
140        label.addStyleName("active");
141    }
142
143    @Override
144    public String getPlaceholder() {
145        return placeholder;
146    }
147
148    @Override
149    public void setPlaceholder(String placeholder) {
150        this.placeholder = placeholder;
151        lblName.setText(placeholder);
152    }
153
154    @Override
155    public HandlerRegistration addKeyDownHandler(KeyDownHandler handler) {
156        return addDomHandler(handler, KeyDownEvent.getType());
157    }
158
159    @Override
160    public HandlerRegistration addKeyUpHandler(KeyUpHandler handler) {
161        return addDomHandler(handler, KeyUpEvent.getType());
162    }
163
164    @Override
165    public HandlerRegistration addKeyPressHandler(KeyPressHandler handler) {
166        return addDomHandler(handler, KeyPressEvent.getType());
167    }
168
169    @Override
170    public HandlerRegistration addChangeHandler(ChangeHandler handler) {
171        return addDomHandler(handler, ChangeEvent.getType());
172    }
173
174    @Override
175    public void setError(String error) {
176        errorMixin.setError(error);
177        setInvalid();
178    }
179
180    @Override
181    public void setSuccess(String success) {
182        errorMixin.setSuccess(success);
183        setValid();
184    }
185
186    @Override
187    public MaterialIcon getIcon() {
188        return icon;
189    }
190
191    @Override
192    public void setIconType(IconType iconType) {
193        icon.setIconType(iconType);
194        icon.setIconPrefix(true);
195    }
196
197    @Override
198    public void setIconPosition(IconPosition position) {
199        icon.setIconPosition(position);
200    }
201
202    @Override
203    public void setIconSize(IconSize size) {
204        icon.setIconSize(size);
205    }
206
207    @Override
208    public void setIconFontSize(double size, Style.Unit unit) {
209        icon.setIconFontSize(size, unit);
210    }
211
212    @Override
213    public void setIconColor(String iconColor) {
214        icon.setIconColor(iconColor);
215    }
216
217    @Override
218    public void setIconPrefix(boolean prefix) {
219        icon.setIconPrefix(prefix);
220    }
221
222    @Override
223    public boolean isIconPrefix() {
224        return icon.isIconPrefix();
225    }
226}
