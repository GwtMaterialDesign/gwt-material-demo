057//@formatter:on
058public class MaterialImage extends MaterialWidget implements HasCaption, HasType<ImageType>, HasImage, HasClickHandlers,
059        HasDoubleClickHandlers, HasAllMouseHandlers, HasLoadHandlers, HasErrorHandlers, HasAllDragAndDropHandlers,
060        HasAllGestureHandlers, HasAllTouchHandlers {
061
062    private final CssTypeMixin<ImageType, MaterialImage> typeMixin = new CssTypeMixin<>(this);
063    private final ImageMixin<MaterialImage> imageMixin = new ImageMixin<>(this);
064
065    /**
066     * Creates an empty image.
067     */
068    public MaterialImage() {
069        super(Document.get().createImageElement());
070    }
071
072    /**
073     * Creates a simple image.
074     */
075    public MaterialImage(String url) {
076        this();
077        setUrl(url);
078    }
079
080    /**
081     * Creates an image with Specific type.
082     */
083    public MaterialImage(String url, ImageType type) {
084        this(url);
085        setType(type);
086    }
087
088    /**
089     * Creates an image from an ImageResource.
090     */
091    public MaterialImage(ImageResource resource) {
092        this();
093        setResource(resource);
094    }
095
096    /**
097     * Creates an image from an ImageResource with Specific type.
098     */
099    public MaterialImage(ImageResource resource, ImageType type) {
100        this(resource);
101        setType(type);
102    }
103
104    @Override
105    public void onLoad() {
106        super.onLoad();
107
108        addStyleName("responsive-img");
109        onInitMaterialDesign();
110    }
111
112    @Override
113    public void setType(ImageType type) {
114        typeMixin.setType(type);
115    }
116
117    @Override
118    public ImageType getType() {
119        return typeMixin.getType();
120    }
121
122    @Override
123    public String getCaption() {
124        return getElement().getAttribute("caption");
125    }
126
127    @Override
128    public void setCaption(String caption) {
129        getElement().setAttribute("data-caption", caption);
130    }
131
132    public native void onInitMaterialDesign() /*-{
133        $wnd.jQuery(document).ready(function(){
134            $wnd.jQuery('.materialboxed').materialbox();
135        });
136    }-*/;
137
138    @Override
139    public void setUrl(String url) {
140        imageMixin.setUrl(url);
141    }
142
143    @Override
144    public String getUrl() {
145        return imageMixin.getUrl();
146    }
147
148    @Override
149    public void setResource(ImageResource resource) {
150        imageMixin.setResource(resource);
151    }
152
153    @Override
154    public ImageResource getResource() {
155        return imageMixin.getResource();
156    }
157
158    public int getWidth() {
159        ImageElement imageElement = getElement().cast();
160        return imageElement.getWidth();
161    }
162
163    public int getHeight() {
164        ImageElement imageElement = getElement().cast();
165        return imageElement.getHeight();
166    }
