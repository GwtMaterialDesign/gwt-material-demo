022import com.google.gwt.dom.client.Document;
023import com.google.gwt.dom.client.Style;
024import com.google.gwt.event.shared.HandlerRegistration;
025import com.google.gwt.user.client.ui.HasValue;
026import com.google.gwt.user.client.ui.Widget;
027import gwt.material.design.client.base.*;
028import gwt.material.design.client.base.helper.UiHelper;
029import gwt.material.design.client.base.mixin.ActiveMixin;
030import gwt.material.design.client.base.mixin.CssTypeMixin;
031import gwt.material.design.client.base.mixin.ToggleStyleMixin;
032import gwt.material.design.client.constants.CollectionType;
033import gwt.material.design.client.constants.CssName;
034import gwt.material.design.client.js.JsMaterialElement;
035
036//@formatter:off
037
038/**
039 * Collection element to define every items
040 *
041 * @author kevzlou7979
042 * @author Ben Dol
043 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#collections">Material Collections</a>
044 */
045//@formatter:on
046public class MaterialCollectionItem extends MaterialWidget implements HasDismissible, HasAvatar, HasType<CollectionType>, HasActive {
047
048    private final ToggleStyleMixin<MaterialCollectionItem> avatarMixin = new ToggleStyleMixin<>(this, CssName.AVATAR);
049    private final ToggleStyleMixin<MaterialCollectionItem> dismissableMixin = new ToggleStyleMixin<>(this, CssName.DISMISSABLE);
050    private final CssTypeMixin<CollectionType, MaterialCollectionItem> typeMixin = new CssTypeMixin<>(this);
051    private final ActiveMixin<MaterialCollectionItem> activeMixin = new ActiveMixin<>(this);
052
053    private HandlerRegistration handlerReg;
054
055    public MaterialCollectionItem() {
056        super(Document.get().createLIElement(), CssName.COLLECTION_ITEM);
057        UiHelper.addMousePressedHandlers(this);
058    }
059
060    @Override
061    protected void onLoad() {
062        super.onLoad();
063        initDismissableCollection();
064    }
065
066    protected void initDismissableCollection() {
067        JsMaterialElement.initDismissableCollection();
068    }
069
070    @Override
071    public void setType(CollectionType type) {
072        typeMixin.setType(type);
073        if (type == CollectionType.CHECKBOX) {
074            applyCheckBoxType();
075        }
076    }
077
078    private void applyCheckBoxType() {
079        if (getWidgetCount() > 0) {
080            getWidget(0).getElement().getStyle().setDisplay(Style.Display.INLINE);
081        }
082        if (handlerReg != null) {
083            handlerReg.removeHandler();
084            handlerReg = null;
085        }
086        handlerReg = addClickHandler(event -> {
087            for (Widget w : MaterialCollectionItem.this) {
088                if (w instanceof MaterialCollectionSecondary) {
089                    for (Widget a : (MaterialCollectionSecondary) w) {
090                        if (a instanceof HasValue) {
091                            try {
092                                @SuppressWarnings("unchecked")
093                                HasValue<Boolean> cb = (HasValue<Boolean>) a;
094                                if (cb.getValue()) {
095                                    cb.setValue(false);
096                                } else {
097                                    cb.setValue(true);
098                                }
099                            } catch (ClassCastException ex) {
100                                // Ignore non-boolean has value handlers.
101                            }
102                        }
103                    }
104                }
105            }
106        });
107    }
108
109    @Override
110    public CollectionType getType() {
111        return typeMixin.getType();
112    }
113
114    @Override
115    public void setDismissible(boolean dismissible) {
116        dismissableMixin.setOn(dismissible);
117    }
118
119    @Override
120    public boolean isDismissible() {
121        return dismissableMixin.isOn();
122    }
123
124    @Override
125    public void setAvatar(boolean avatar) {
126        avatarMixin.setOn(avatar);
127    }
128
129    @Override
130    public boolean isAvatar() {
131        return avatarMixin.isOn();
132    }
133
134    @Override
135    public void setActive(boolean active) {
136        activeMixin.setActive(active);
137    }
138
139    @Override
140    public boolean isActive() {
141        return activeMixin.isActive();
142    }
143}
