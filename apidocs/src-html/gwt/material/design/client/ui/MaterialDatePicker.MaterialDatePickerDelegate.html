041 * <h3>UiBinder Usage:</h3>

086    @Override
087    protected void onAttach() {
088        super.onAttach();
089        id = String.valueOf(hashCode());
090        this.clear();
091        panel = new HTMLPanel("<input placeholder='"+placeholder+"' type='date' id='"+id+"' class='datepicker'>");
092        this.add(panel);
093        panel.addStyleName(orientation.getCssName());
094        initDatePicker(id, selectionType.name(), this);
095        initClickHandler(id, this);
096        panel.add(lblError);
097    }

148    /**
149     * A delegate which implements handling of events from date picker.
150     * @param delegate which will be notified on picker events.
151     * @see MaterialDatePickerDelegate
152     */
153    public void setDelegate(MaterialDatePickerDelegate delegate) {
154        this.delegate = delegate;
155    }
156
157    void notifyDelegate() {
158        if(delegate != null) {
159            delegate.onCalendarClose(getDate());
160        }
161    }

163    public Date getDate() {
164        return getPickerDate();
165    }
166
167    public native void clearValues() /*-{
168        $wnd.jQuery('.datepicker').val("");
169    }-*/;
170
171    public Date getPickerDate() {
172        try {
173            DateTimeFormat sdf = DateTimeFormat.getFormat("d MMM, yyyy");
174            date = sdf.parse(getDatePickerValue(id));
175            return date;
176        } catch(Exception e) {
177            e.printStackTrace();
178            return null;
179        }
180    }
181
182    /**
183     * Sets the current date of the picker.
184     * @param date - must not be <code>null</code>
185     */
186    public void setDate(Date date) {
187        if(date == null) {
188            return;
189        }
190        this.date = date;
191        initDatePicker(id, selectionType.name(), this);
192        DateTimeFormat sdf = DateTimeFormat.getFormat("d MMM, yyyy");
193        setDatePickerValue(sdf.format(date), id);
194        /*DateTimeFormat sdfSetter = DateTimeFormat.getFormat("yyyy-MM-dd");
195        selectDate(sdfSetter.format(date), id, this);*/
196    }
