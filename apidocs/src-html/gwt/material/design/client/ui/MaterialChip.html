020package gwt.material.design.client.ui;
021
022import com.google.gwt.dom.client.Document;
023import com.google.gwt.dom.client.Style;
024import com.google.gwt.resources.client.ImageResource;
025import com.google.gwt.user.client.ui.Image;
026import gwt.material.design.client.base.HasIcon;
027import gwt.material.design.client.base.HasImage;
028import gwt.material.design.client.base.HasLetter;
029import gwt.material.design.client.base.MaterialWidget;
030import gwt.material.design.client.base.mixin.LetterMixin;
031import gwt.material.design.client.constants.*;
032import gwt.material.design.client.ui.html.Span;
033
034//@formatter:off
035
036/**
037 * Chips can be used to represent small blocks of information.
038 * They are most commonly used either for contacts or for tags.
039 * <p>
040 * <h3>UiBinder Usage:</h3>
041 * <pre>
042 * {@code// Simple Chips
043 * <m:MaterialChip text="Default" iconType="close"/>
044 *
045 * // Static Chip
046 * <m:MaterialChip text="Apple" />
047 *
048 * // Contact Chips
049 * <m:MaterialChip url="http://b.vimeocdn.com/ps/339/488/3394886_300.jpg" text="Yunalis Mat Zara'ai" iconType="CLOSE"/>
050 * }
051 * </pre>
052 *
053 * @author kevzlou7979
054 * @author Ben Dol
055 * @see <a href="http://gwtmaterialdesign.github.io/gwt-material-demo/#!chips">Material Chips</a>
056 */
057//@formatter:on
058public class MaterialChip extends MaterialWidget implements HasImage, HasIcon, HasLetter {
059
060    private MaterialIcon icon = new MaterialIcon();
061    private Span span = new Span();
062
063    private ImageResource resource;
064    private Image image = new Image();
065
066    private final LetterMixin<MaterialChip> letterMixin = new LetterMixin<>(this);
067
068    /**
069     * Creates an empty chip.
070     */
071    public MaterialChip() {
072        super(Document.get().createDivElement(), CssName.CHIP);
073    }
074
075    public MaterialChip(String text) {
076        this();
077        setText(text);
078    }
079
080    public MaterialChip(String text, IconType icon) {
081        this(text);
082        setIconType(icon);
083    }
084
085    public MaterialChip(String text, String imageUrl) {
086        this(text);
087        setUrl(imageUrl);
088    }
089
090    public MaterialChip(String text, Color bgColor, Color textColor) {
091        this(text);
092        setBackgroundColor(bgColor);
093        setTextColor(textColor);
094    }
095
096    @Deprecated
097    public MaterialChip(String text, String bgColor, String textColor) {
098        this(text, Color.fromStyleName(bgColor), Color.fromStyleName(textColor));
099    }
100
101    public void setText(String text) {
102        span.setText(text);
103        add(span);
104    }
105
106    public String getText() {
107        return span.getElement().getInnerText();
108    }
109
110    @Override
111    public void setUrl(String url) {
112        image.setUrl(url);
113        add(image);
114    }
115
116    @Override
117    public String getUrl() {
118        return image.getUrl();
119    }
120
121    @Override
122    public void setResource(ImageResource resource) {
123        this.resource = resource;
124        image.setResource(resource);
125        add(image);
126    }
127
128    @Override
129    public ImageResource getResource() {
130        return resource;
131    }
132
133    /**
134     * @return the image
135     */
136    public Image getImage() {
137        return image;
138    }
139
140    /**
141     * @param image the image to set
142     */
143    public void setImage(Image image) {
144        this.image = image;
145    }
146
147    @Override
148    public MaterialIcon getIcon() {
149        return icon;
150    }
151
152    @Override
153    public void setIconType(IconType iconType) {
154        icon.setIconType(iconType);
155        add(icon);
156    }
157
158    @Override
159    public void setIconPosition(IconPosition position) {
160        icon.setIconPosition(position);
161    }
162
163    @Override
164    public void setIconSize(IconSize size) {
165        icon.setIconSize(size);
166    }
167
168    @Override
169    public void setIconFontSize(double size, Style.Unit unit) {
170        icon.setIconFontSize(size, unit);
171    }
172
173    @Override
174    public void setIconColor(Color iconColor) {
175        icon.setIconColor(iconColor);
176    }
177
178    @Override
179    public void setIconPrefix(boolean prefix) {
180        icon.setIconPrefix(prefix);
181    }
182
183    @Override
184    public boolean isIconPrefix() {
185        return icon.isIconPrefix();
186    }
187
188    @Override
189    public void setLetter(String letter) {
190        letterMixin.setLetter(letter);
191    }
192
193    @Override
194    public String getLetter() {
195        return letterMixin.getLetter();
196    }
197
198    @Override
199    public void setLetterColor(Color letterColor) {
200        letterMixin.setLetterColor(letterColor);
201    }
202
203    @Override
204    public void setLetterBackgroundColor(Color letterBackgroundColor) {
205        letterMixin.setLetterBackgroundColor(letterBackgroundColor);
206    }
207
208    public LetterMixin<MaterialChip> getLetterMixin() {
209        return letterMixin;
210    }
211}
