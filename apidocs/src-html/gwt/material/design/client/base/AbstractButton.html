020package gwt.material.design.client.base;
021
022import com.google.gwt.dom.client.Element;
023import com.google.gwt.dom.client.Style;
024import com.google.gwt.user.client.History;
025import gwt.material.design.client.base.mixin.ActivatesMixin;
026import gwt.material.design.client.base.mixin.CssTypeMixin;
027import gwt.material.design.client.constants.*;
028import gwt.material.design.client.ui.html.Span;
029
030/**
031 * @author Ben Dol
032 */
033public abstract class AbstractButton extends MaterialWidget implements HasHref, HasGrid, HasActivates,
034        HasTargetHistoryToken, HasType<ButtonType> {
035
036    private final ActivatesMixin<AbstractButton> activatesMixin = new ActivatesMixin<>(this);
037    private final CssTypeMixin<ButtonType, AbstractButton> cssTypeMixin = new CssTypeMixin<>(this);
038
039    private Span span = new Span();
040    private ButtonSize size;
041
042    private String targetHistoryToken;
043
044    /**
045     * Creates button with RAISED type.
046     */
047    protected AbstractButton() {
048        setElement(createElement());
049
050        getElement().getStyle().setCursor(Style.Cursor.POINTER);
051    }
052
053    protected AbstractButton(String... initialClass) {
054        this();
055        setInitialClasses(initialClass);
056    }
057
058    protected AbstractButton(String text, Color bgColor, WavesType waves) {
059        this(null, text, bgColor);
060        setWaves(waves);
061    }
062
063    protected AbstractButton(final ButtonType type, String text, Color bgColor, WavesType waves) {
064        this(type, text, bgColor);
065        setWaves(waves);
066    }
067
068    protected AbstractButton(final ButtonType type, String text, Color bgColor) {
069        this(type, text);
070        setBackgroundColor(bgColor);
071    }
072
073    protected AbstractButton(final ButtonType type, String text) {
074        this(type);
075        setText(text);
076    }
077
078    protected AbstractButton(final ButtonType type) {
079        this();
080        setType(type);
081    }
082
083    protected abstract Element createElement();
084
085    @Override
086    public void setHref(String href) {
087        getElement().setAttribute("href", href);
088    }
089
090    @Override
091    public String getHref() {
092        return getElement().getAttribute("href");
093    }
094
095    @Override
096    public void setTarget(String target) {
097        getElement().setAttribute("target", target);
098    }
099
100    @Override
101    public String getTarget() {
102        return getElement().getAttribute("target");
103    }
104
105    @Override
106    public void setActivates(String activates) {
107        removeStyleName(getActivates() + " " + CssName.DROPDOWN_BUTTON);
108        activatesMixin.setActivates(activates);
109        addStyleName(activates + " " + CssName.DROPDOWN_BUTTON);
110    }
111
112    @Override
113    public String getActivates() {
114        return activatesMixin.getActivates();
115    }
116
117    @Override
118    public void setType(ButtonType type) {
119        cssTypeMixin.setType(type);
120    }
121
122    @Override
123    public ButtonType getType() {
124        return cssTypeMixin.getType();
125    }
126
127    /**
128     * Set the buttons size.
129     */
130    public void setSize(ButtonSize size) {
131        if (this.size != null) {
132            removeStyleName(this.size.getCssName());
133        }
134        this.size = size;
135
136        if (size != null) {
137            addStyleName(size.getCssName());
138        }
139    }
140
141    /**
142     * Get the buttons size.
143     */
144    public ButtonSize getSize() {
145        return size;
146    }
147
148    /**
149     * Get the buttons span text.
150     */
151    public String getText() {
152        return span.getText();
153    }
154
155    /**
156     * Set the buttons span text.
157     */
158    public void setText(String text) {
159        span.setText(text);
160
161        if (!span.isAttached()) {
162            add(span);
163        }
164    }
165
166    /**
167     * Set the target history token for the widget. Note, that you should use either
168     * {@link #setTargetHistoryToken(String)} or {@link #setHref(String)}, but not both as
169     * {@link #setHref(String)} resets the target history token.
170     *
171     * @param targetHistoryToken String target history token of the widget
172     */
173    @Override
174    public void setTargetHistoryToken(final String targetHistoryToken) {
175        this.targetHistoryToken = targetHistoryToken;
176        if (targetHistoryToken != null) {
177            setHref("#" + History.encodeHistoryToken(targetHistoryToken));
178        }
179    }
180
181    /**
182     * Get the target history token for the widget. May return {@code null} if no
183     * history token has been set or if it has been reset by {@link #setHref(String)}.
184     *
185     * @return String the widget's target history token.
186     */
187    @Override
188    public String getTargetHistoryToken() {
189        return targetHistoryToken;
190    }
191
192    public Span getSpan() {
193        return span;
194    }
195}
