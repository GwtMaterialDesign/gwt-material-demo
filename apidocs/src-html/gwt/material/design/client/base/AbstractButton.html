023import com.google.gwt.dom.client.Element;
024import com.google.gwt.dom.client.Style;
025import com.google.gwt.event.dom.client.ClickEvent;
026import com.google.gwt.event.dom.client.ClickHandler;
027import com.google.gwt.event.dom.client.DoubleClickEvent;
028import com.google.gwt.event.dom.client.DoubleClickHandler;
029import com.google.gwt.event.dom.client.HasAllMouseHandlers;
030import com.google.gwt.event.dom.client.HasClickHandlers;
031import com.google.gwt.event.dom.client.HasDoubleClickHandlers;
032import com.google.gwt.event.dom.client.MouseDownEvent;
033import com.google.gwt.event.dom.client.MouseDownHandler;
034import com.google.gwt.event.dom.client.MouseMoveEvent;
035import com.google.gwt.event.dom.client.MouseMoveHandler;
036import com.google.gwt.event.dom.client.MouseOutEvent;
037import com.google.gwt.event.dom.client.MouseOutHandler;
038import com.google.gwt.event.dom.client.MouseOverEvent;
039import com.google.gwt.event.dom.client.MouseOverHandler;
040import com.google.gwt.event.dom.client.MouseUpEvent;
041import com.google.gwt.event.dom.client.MouseUpHandler;
042import com.google.gwt.event.dom.client.MouseWheelEvent;
043import com.google.gwt.event.dom.client.MouseWheelHandler;
044import com.google.gwt.event.shared.HandlerRegistration;
045import com.google.gwt.user.client.History;
046import gwt.material.design.client.base.mixin.ActivatesMixin;
047import gwt.material.design.client.base.mixin.CssTypeMixin;
048import gwt.material.design.client.constants.ButtonSize;
049import gwt.material.design.client.constants.ButtonType;
050import gwt.material.design.client.constants.WavesType;
051import gwt.material.design.client.ui.html.Span;
052
053/**
054 * @author Ben Dol
055 */
056public abstract class AbstractButton extends MaterialWidget implements HasHref, HasGrid, HasActivates,
057        HasTargetHistoryToken, HasType<ButtonType>, HasClickHandlers, HasAllMouseHandlers,
058        HasDoubleClickHandlers {
059
060    private final ActivatesMixin<AbstractButton> activatesMixin = new ActivatesMixin<>(this);
061    private final CssTypeMixin<ButtonType, AbstractButton> cssTypeMixin = new CssTypeMixin<>(this);
062
063    private Span span = new Span();
064    private ButtonSize size;
065
066    private String targetHistoryToken;
067
068    /** Creates button with RAISED type.
069     */
070    protected AbstractButton() {
071        setElement(createElement());
072
073        getElement().getStyle().setCursor(Style.Cursor.POINTER);
074    }
075
076    protected AbstractButton(String text, String bgColor, WavesType waves) {
077        this(null, text, bgColor);
078        setWaves(waves);
079    }
080
081    protected AbstractButton(final ButtonType type, String text, String bgColor, WavesType waves) {
082        this(type, text, bgColor);
083        setWaves(waves);
084    }
085
086    protected AbstractButton(final ButtonType type, String text, String bgColor) {
087        this(type, text);
088        setBackgroundColor(bgColor);
089    }
090
091    protected AbstractButton(final ButtonType type, String text) {
092        this(type);
093        setText(text);
094    }
095
096    protected AbstractButton(final ButtonType type) {
097        this();
098        setType(type);
099    }
100
101    protected abstract Element createElement();
102
103    @Override
104    public void setHref(String href) {
105        getElement().setAttribute("href", href);
106    }
107
108    @Override
109    public String getHref() {
110        return getElement().getAttribute("href");
111    }
112
113    @Override
114    public void setTarget(String target) {
115        getElement().setAttribute("target", target);
116    }
117
118    @Override
119    public String getTarget() {
120        return getElement().getAttribute("target");
121    }
122
123    @Override
124    public void setActivates(String activates) {
125        removeStyleName(getActivates() + " dropdown-button");
126        activatesMixin.setActivates(activates);
127        addStyleName(activates + " dropdown-button");
128    }
129
130    @Override
131    public String getActivates() {
132        return activatesMixin.getActivates();
133    }
134
135    @Override
136    public void setType(ButtonType type) {
137        cssTypeMixin.setType(type);
138    }
139
140    @Override
141    public ButtonType getType() {
142        return cssTypeMixin.getType();
143    }
144
145    public void setSize(ButtonSize size) {
146        if(this.size != null) {
147            removeStyleName(this.size.getCssName());
148        }
149        this.size = size;
150
151        if(size != null) {
152            addStyleName(size.getCssName());
153        }
154    }
155
156    public ButtonSize getSize() {
157        return size;
158    }
159
160    public String getText() {
161        return span.getText();
162    }
163
164    public void setText(String text) {
165        span.setText(text);
166        add(span);
167    }
168
169    /**
170     * Set the target history token for the widget. Note, that you should use either
171     * {@link #setTargetHistoryToken(String)}or {@link #setHref(String)}, but not both as
172     * {@link #setHref(String)} resets the target history token.
173     * @param targetHistoryToken String target history token of the widget
174     */
175    @Override
176    public void setTargetHistoryToken(final String targetHistoryToken) {
177        this.targetHistoryToken = targetHistoryToken;
178        if (targetHistoryToken != null) {
179            setHref("#" + History.encodeHistoryToken(targetHistoryToken));
180        }
181    }
182
183    /**
184     * Get the target history token for the widget. May return {@code null} if no
185     * history token has been set or if it has been reset by {@link #setHref(String)}.
186     * @return String the widget's target history token.
187     */
188    @Override
189    public String getTargetHistoryToken() {
190        return targetHistoryToken;
191    }
192
193    @Override
194    public HandlerRegistration addClickHandler(ClickHandler handler) {
195        return addDomHandler(handler, ClickEvent.getType());
196    }
197
198    @Override
199    public HandlerRegistration addMouseDownHandler(final MouseDownHandler handler) {
200        return addDomHandler(handler, MouseDownEvent.getType());
201    }
202
203    @Override
204    public HandlerRegistration addMouseMoveHandler(final MouseMoveHandler handler) {
205        return addDomHandler(handler, MouseMoveEvent.getType());
206    }
207
208    @Override
209    public HandlerRegistration addMouseOutHandler(final MouseOutHandler handler) {
210        return addDomHandler(handler, MouseOutEvent.getType());
211    }
212
213    @Override
214    public HandlerRegistration addMouseOverHandler(final MouseOverHandler handler) {
215        return addDomHandler(handler, MouseOverEvent.getType());
216    }
217
218    @Override
219    public HandlerRegistration addMouseUpHandler(final MouseUpHandler handler) {
220        return addDomHandler(handler, MouseUpEvent.getType());
221    }
222
223    @Override
224    public HandlerRegistration addMouseWheelHandler(final MouseWheelHandler handler) {
225        return addDomHandler(handler, MouseWheelEvent.getType());
226    }
227
228    @Override
229    public HandlerRegistration addDoubleClickHandler(DoubleClickHandler handler) {
230        return addDomHandler(handler, DoubleClickEvent.getType());
231    }
232}
